async function loadQs(){
  var gameDate = getGameDate();

  if(DB.questions&&DB.questions[ck]){
    var c=DB.questions[ck], v=validateQs(c);

  var prompt =
    'You are an expert NFL historian and question writer. Generate exactly 5 '+SITE.team+' trivia questions for '+gameDate+'.\n\n'
    +'QUESTION MIX — follow this exactly:\n'
    +'Q1 (easy): Evergreen team history — founding, championships, iconic moments, legends\n'

    +'- Asks for a specific rushing/receiving yard total on Q1 or Q2\n'
    +'- Is ambiguous or has multiple defensible correct answers\n\n'
    +'Return ONLY valid JSON array, no markdown:\n'

  try{
    var res=await fetch('https://api.anthropic.com/v1/messages',{
      method:'POST',headers:{'Content-Type':'application/json'},
      body:JSON.stringify({model:'claude-sonnet-4-6',max_tokens:1400,messages:[{role:'user',content:prompt}]})
    });
    if(!res.ok) throw new Error('API '+res.status);
    var data=await res.json();
    var txt=data.content.map(function(b){return b.text||'';}).join('').replace(/```json|```/g,'').trim();
    var m=txt.match(/\[[\s\S]*\]/);
    if(!m) throw new Error('no json');
    var p=JSON.parse(m[0]);
    if(!Array.isArray(p)||p.length<5) throw new Error('bad array');
    p=validateQs(p);
    if(p.length<5) throw new Error('validation');
    DB.questions[ck]=p; saveDB(); return p;
  }catch(e){ console.warn('AI failed:',e); return fallbackQs(); }
}

function validateQs(arr){
  return arr.filter(function(q){
    if(!q.q||!Array.isArray(q.o)||q.o.length<4) return false;
    if(q.a<0||q.a>=q.o.length) return false;
    var lower = q.o.map(function(o){return o.toString().trim().toLowerCase();});
    if(lower.indexOf('all of the above')>=0||lower.indexOf('none of the above')>=0) return false;
    var seen={};
    for(var i=0;i<lower.length;i++){ if(seen[lower[i]]) return false; seen[lower[i]]=true; }
    return true;
  });
}

function pickAnswer(idx){
  clearInterval(tmr);
  var q=qs[qIdx];
  var btns=document.querySelectorAll('.ob');
  btns.forEach(function(b){b.disabled=true;});
  var correct=idx===q.a;
  if(correct){
    btns[idx].classList.add('ok');
    var earned=calcPts(timeLeft);
    score=clamp(score+earned,0,9999);
    qResults.push({correct:true,picked:idx,timedOut:false,earned:earned,penalty:0});
    updatePip(qIdx,'ok');
    showFb('ok','✅ Correct! '+(earned===MAX_PER_Q?'⚡ Full points!':''+earned+'/'+MAX_PER_Q+' pts')+' '+q.f, earned);
  } else {
    btns[idx].classList.add('no'); btns[q.a].classList.add('ok');
    var pen=PENALTY_WRONG;
    score=Math.max(0,score+pen);
    qResults.push({correct:false,picked:idx,timedOut:false,earned:0,penalty:pen});
    updatePip(qIdx,'no');
    document.getElementById('flagBtn').classList.add('show');
    showFb('no','❌ Wrong! '+q.f, pen);
  }
  qIdx++;
  setNxt(qIdx>=5?'See Results →':'Next question →');
}

// ─── PERSIST ──────────────────────────────────────────────────────────────────
function persistScore(uid,sc,res,qset){
  var dk=SITE.teamId+'_'+getGameDate();
  var day=dbG('daily',dk)||{};
  day[uid]={score:sc,results:res,correct:res.filter(function(r){return r.correct;}).length};
  dbS('daily',dk,day);

  var sk=SITE.teamId+'_'+uid;
  var streak=dbG('streaks',sk)||{currentPerfect:0,longestPerfect:0,play:0,allTimeScore:0,gamesPlayed:0,perfectGames:0};
  var perfect=res.filter(function(r){return r.correct;}).length===5;
  streak.allTimeScore=(streak.allTimeScore||0)+sc;
  streak.gamesPlayed=(streak.gamesPlayed||0)+1;
  if(perfect){
    streak.currentPerfect=(streak.currentPerfect||0)+1;
    streak.longestPerfect=Math.max(streak.longestPerfect||0,streak.currentPerfect);
    streak.perfectGames=(streak.perfectGames||0)+1;
  } else {
    streak.currentPerfect=0;
  }
  dbS('streaks',sk,streak);

  var net=dbG('network',SITE.teamId)||{team:SITE.team,total:0,games:0};
  net.team=SITE.team; net.total+=sc; net.games++; net.avg=Math.round(net.total/net.games);
  dbS('network',SITE.teamId,net);

  saveDB();
}

function copyText(txt,btn,reset){
  function done(){btn.textContent='✅ Copied!';setTimeout(function(){btn.textContent=reset;},2000);}
  if(navigator.clipboard&&navigator.clipboard.writeText){navigator.clipboard.writeText(txt).then(done).catch(fb2);}else{fb2();}
  function fb2(){var ta=document.createElement('textarea');ta.value=txt;ta.style.cssText='position:fixed;opacity:0';document.body.appendChild(ta);ta.select();try{document.execCommand('copy');done();}catch(e){}document.body.removeChild(ta);}
}
